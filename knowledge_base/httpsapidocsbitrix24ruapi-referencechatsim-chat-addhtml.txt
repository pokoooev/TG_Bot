## ПРИМЕРЫ КОДА

Мой новый закрытый чат
---

Добро пожаловать в чат
---

$result = restCommand(
    'im.chat.add',
    Array(
        'TYPE' => 'CHAT',
        'TITLE' => 'Мой новый закрытый чат',
        'DESCRIPTION' => 'Очень важный чат',
        'COLOR' => 'PINK',
        'MESSAGE' => 'Добро пожаловать в чат',
        'USERS' => Array(1,2),
        'AVATAR' => 'base64 image',
        'ENTITY_TYPE' => 'CHAT',
        'ENTITY_ID' => 13,
        'OWNER_ID' => 39,
    ),
    $_REQUEST["auth"]
);
---
