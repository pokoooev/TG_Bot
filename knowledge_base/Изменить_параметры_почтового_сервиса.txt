# Изменить параметры почтового сервиса
URL: https://apidocs.bitrix24.ru/api-reference/mailservice/mailservice-update.html

## ОПИСАНИЕ
Scope: mailservice Кто может выполнять метод: любой пользователь Метод mailservice.update обновляет параметры почтового сервиса. Параметр / Тип Описание С версии ID
unknown Идентификатор почтового сервиса ACTIVE
unknown Активность сервиса (Y / N) NAME
unknown Имя добавляемого почтового сервиса SERVER
unknown Адрес добавляемого сервера PORT
unknown Номер порта ENCRYPTION
unknown Необходимость шифрования (Y / N) LINK
unknown Ссылка на почтовый сервис SORT
unknown Индекс сортировки Как использовать примеры в документации

## ПРИМЕРЫ КОДА

try
{
 const response = await $b24.callMethod(
  "mailservice.update",
  {
   'ID': 5,
   'ACTIVE': 'N',
   'NAME': 'Почтовый сервис Yandex',
   'SERVER': 'imap.yandex.ru',
   'PORT': '993',
   'ENCRYPTION': 'Y',
   'LINK': 'https://mail.yandex.ru/',
   'SORT': '666'
  }
 );
 
 const result = response.getData().result;
 console.info(result);
}
catch(error)
{
 console.error(error);
}
